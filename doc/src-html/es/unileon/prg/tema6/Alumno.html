001    package es.unileon.prg.tema6;
002    /**
003     * Clase que representa un alumno.
004     * 
005     * @author PRG
006     * @version 1.0
007     */
008    public class Alumno {
009            /**
010             * Identificador del alumno
011             * 
012             * @see String
013             */
014            private String _niu;
015            /**
016             * Nombre del alumno
017             * 
018             * @see String
019             */
020            private String _nombre;
021            /**
022             * Nota del alumno
023             */
024            private float _nota;
025    
026            /**
027             * Constructor de la clase. Crea un alumno con la informacion recibida
028             * 
029             * @param niu
030             *            Identificador del alumno
031             * @param nombre
032             *            Nombre del alumno
033             */
034            public Alumno(String niu, String nombre) {
035                    _niu = niu;
036                    _nombre = nombre;
037            }
038    
039            /**
040             * Asigna la calificacion del alumno
041             * 
042             * @param nota
043             *            Nota del alumno
044             */
045            public void asignarNota(float nota) {
046                    _nota = nota;
047            }
048    
049            
050            
051            //Inicio modificacion Apartado 030101 - ejercicio01
052            public float obtenerNota()
053            {
054                    return _nota;
055            }
056            
057            //Fin modificacion Apartado 030101 - ejercicio01
058            
059            
060            
061            
062            /*
063             * (non-Javadoc)
064             * @see java.lang.Object#toString()
065             */
066            
067            public String toString() {
068                    StringBuffer salida = new StringBuffer();
069                    
070                    salida.append("NIU: "+_niu+" ");
071                    salida.append("Nombre: "+_nombre+" ");
072                    salida.append("Nota: "+_nota+" ");
073                    
074                    return salida.toString();
075            }
076            
077    }
