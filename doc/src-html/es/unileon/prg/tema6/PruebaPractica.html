001    package es.unileon.prg.tema6;
002    import java.lang.reflect.Method;
003    import java.lang.reflect.InvocationTargetException;
004    
005    /**
006     * Clase de prueba de las practicas de ADA
007     *
008     * @author PRG
009     * @version 1.0
010     */
011    public class PruebaPractica {
012    
013            private static String CLASE = "es.unileon.prg.tema6.Apartado";
014            private static String METODO = "ejercicio";
015    
016            /**
017             * Metodo principal de la aplicacion
018             *
019             * @param args
020             *            Parametros de la aplicacion. Recibe como parametro el apartado
021             *            y el ejercicio a ejecutar
022             */
023            public static void main(String[] args) {
024                    if (args.length != 2) {
025                            System.out.println();
026                            System.out.println("Numero de parametros erroneo");
027                            System.out.println("Formato: ");
028                            System.out.println("PruebaPractica [apartado] [ejercicio]");
029                            System.out.println();
030                    } else {
031                            String nombreClase = CLASE + args[0];
032                            String nombreMetodo = METODO + args[1];
033    
034                            try {
035                                    Class<?> clase = Class.forName(nombreClase);
036                                    Method metodo = clase.getMethod(nombreMetodo);
037                                    metodo.invoke(clase.newInstance());
038                            } catch (ClassNotFoundException classNotFound) {
039                                    System.out.println();
040                                    System.out
041                                                    .println("Codigo de apartado incorrecto. Por favor, revise el enunciado de la practica");
042                                    System.out.println();
043                            } catch (InvocationTargetException invocation) {
044                                    System.out.println();
045                                    invocation.printStackTrace();
046                                    System.out.println();
047                            } catch (Exception e) {
048                                    System.out.println();
049                                    System.out
050                                                    .println("Numero de ejercicio incorrecto. Por favor, revise el enunciado de la practica");
051                                    System.out.println();
052                            }
053                    }
054            }
055    }
