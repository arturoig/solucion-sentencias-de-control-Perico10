003    /** Clase que representa una tabla de conversion de grados Celsius a Fahrenheit
004     * @author PRG
005     * @version (1.0)
006     */
007    
008    
009    public class TablaConversion
010    {
011            /**
012             * Columnas de la tabla
013             */
014        private int columnas;
015        /**
016             * Filas o lineas de la tabla
017             */
018        private int lineas;
019        /**
020         * Separador entre elemetos de la tabla
021         */
022        final String SEPARADOR = "\t";
023        
024        /**
025             * Constructor
026             * @param columnas
027             *      Numero de columnas de la tabla
028             * @param lineas
029             *      Numero de lineas de la tabla
030             */
031    
032        public TablaConversion(int columnas, int lineas)
033        {
034            this.columnas = columnas;
035            this.lineas = lineas;
036        }
037            
038        /**
039        *  Imprime una tabla de conversion de grados Celsius a Fahrenheit
040        */
041        public void imprimir()
042        {
043    
044            System.out.println("Tabla de conversion de temperaturas de grados Celsius a grados Farenhait");
045            System.out.println();
046           
047            // Imprimir cabecera
048            //Inicio modificacion del ejercicio10 del Apartado030201
049           
050            for (int i = 0; i < columnas; i++)
051                System.out.print("C" + SEPARADOR + "F" + SEPARADOR);
052            System.out.println();
053            
054            
055            //Fin modificacion del ejercicio10 del Apartado030201
056            
057            // Imprimir cada linea  
058            for (int i = 0; i < this.lineas; i++)
059                imprimirLinea(i);
060             
061         }
062    
063    
064        /**
065         * Imprime una linea de la tabla de conversion de grados Celsiuis a
066         * Fahrenheit, utilizando la informacion dada por el numero de
067         * linea actual
068         * @param estaLinea Numero de la linea a imprimir
069         * 
070         * */
071         public void imprimirLinea(int estaLinea)
072         {
073             int c; // temperatura en celsius
074             int f; // temperatura en fahrenheit
075             
076             //Inicio modificacion del ejercicio10 del Apartado030201
077    
078             for (int i = 0; i < this.columnas; i++)
079             {
080                 // Se calcula la temperatura en celsius que corresponde
081                 c = estaLinea * this.columnas + i;
082                 f = fahrenheit(c);
083                 System.out.print(c + SEPARADOR + f + SEPARADOR);
084             }
085             
086             //Fin modificacion del ejercicio10 del Apartado030201
087                      
088             System.out.println();
089         }
090    
091    
092        /**
093         * Transforma una temperatura dada en grados celsius a
094         * fahrenheit
095         * @param celsius Temperatura en grados Celsiuis
096         * @return Temperatura en grados Fahrenheit
097         */
098         public int fahrenheit(int celsius)
099         {
100             return (int)((celsius * 1.8) + 32);
101         }
102    }
